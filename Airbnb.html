Now we want to write helper functions to clean our dataset. The functions will be the following:

  1. clean_str_to_list (df, col)
  2. get_year_from_date (df, date_col)
  3. handle_null_values (df)
  4. clean_listings (listings, column_names)

We detected columns that are categorical but do not contain the items itself, but lists of items. We want to clean this columns and provide clear information which row contains which item. Moreover we detected several date columns that are stored as strings. To make use of these columns in the model we want to extract the year out of the dates and store it as a numerical value.

We saw above, that we have several columns containing null values and we want to have a useful approach to get rid of nulls.

In the end we want to bring all together and generate a new clean listings dataset.

def clean_str_to_list(df, col):
    '''
    INPUT:
    df - the dataframe that contains the column to be cleaned 
    col - the column that contains lists with items 
    
    OUTPUT:
    df_all_items - a dataframe with items as columns and value 1 if row contained this particluar item
    
    Detects all items in lists for every row and combines them to a new set. Uses the clean item list 
    to create a new DataFrame. Instead of one column we obtain new columns foreach item with information 
    whether the item was part of that row.
    '''
    
    #create empty list for all items in column and split column into items
    all_items = []
    split_items_to_list = df[col].apply(lambda x: x[1:-1].split(','))
    
    #clean the string foreach item and add it to list
    for i in range(len(split_items_to_list)):
        item = [a.replace("'",'').strip() for a in split_items_to_list[i]]
        split_items_to_list[i] = item
        all_items.extend(item)
    
    #remove duplicated
    all_items = list(set(all_items))
    
    #remove empty item
    try:
        all_items.remove('')
    except:
        pass
    
    #create array to fill it with 1 where row has the column item in list
    items_array = np.zeros((len(split_items_to_list),len(all_items)))
    
    #fill the array with 1
    for i in range(0,len(split_items_to_list)):
        for j in range(0,len(all_items)):
            if all_items[j] in split_items_to_list[i]:
                items_array[i][j] = 1
    
    #create column name list
    col_names_items = [col + "_" + a.replace('"','') for a in all_items]
    
    #create DataFrame out of array
    df_all_items = pd.DataFrame(items_array, columns = col_names_items)
    
    return df_all_items

def get_year_from_date(df, date_col):
    '''
    INPUT:
    df - the dataframe that contains the column to be cleaned 
    date_col - the column that contains a date as a string
    
    OUTPUT:
    col_to_year - a column holding the year of the date as an int64
    
    Provides a year column that can be used instead of the date column. Due to its dtype it now can be
    used as a numerical variable.
    '''
    
    # fill all NaNs with string nan for a better accessibility and define an int64 nan to be able to convert 
    # the whole column to dtype int64 later on 
    col_to_year = df[date_col].fillna('nan')
    int64_nan = pd.array([np.nan], dtype="Int64")
    
    # extract year from every date and use int64_nan for the NaNs 
    for i in range(len(col_to_year)):
        if col_to_year[i] == 'nan':
            col_to_year[i] = int64_nan[0]
        else:
            col_to_year[i] = np.int64(datetime.strptime(col_to_year[i], '%Y-%m-%d').year)

    return col_to_year

def handle_null_values(df):
    '''
    INPUT:
    df -  - the dataframe that has to be cleaned before using it in a model 
    
    OUTPUT:
    this_df - a dataframe holding no NULL values 
    
    Provides a dataframe that has no NULL values, to use it in our models. Here we distinguish between
    the number of NULL values in a column. If there is less than 1% NULL values, we drop the rows where 
    these particular column values are NULL. If we have more than half of the values NULL, we drop the 
    whole column, because it does not provide any useful information for our prediction. If we have a 
    number of NULLs somewhere in between we impute the values with the mode() strategy.
    '''
    
    this_df = df
    n = df.shape[0]
    columns_contain_null = this_df.isnull().sum()[this_df.isnull().sum() > 1]
    
    delete_rows_with_nans = []
    delete_columns_with_nans = []
    replace_nans = []
    
    for col in columns_contain_null.index:
        if columns_contain_null[col] <= 1/100 * n:
            delete_rows_with_nans.append(col)
        elif columns_contain_null[col] > 1/2 * n:
            delete_columns_with_nans.append(col)
        else:
            replace_nans.append(col)
    
    fill_mean = lambda col: col.fillna(col.mode()[0])
  
    this_df = this_df.dropna(axis=0, subset=delete_rows_with_nans)
    this_df = this_df.drop(delete_columns_with_nans, axis=1)
    
    for col in replace_nans:
        this_df = this_df.apply(fill_mean, axis=0)

    return this_df

def clean_listings(listings, column_names):
    '''
    INPUT:
    listings - the dataframe that has to be cleaned before using it in a model  
    column_names - the column names, we defined before and considered to be useful
    
    OUTPUT:
    listings_rel_cols - a dataframe that is clean, contains only the relevant columns and is ready 
                        to be used in a model
    
    Provides a clean listings dataframe that can be used to understand the most influential coefficients
    in a linear model.
    '''
    
    # initialize new listings df and column names list
    listings_clean = listings
    column_names = column_names
    
    # clean all % and $ characters and convert to float64
    for col in column_names:
        try:
            listings_clean[col] = listings_clean[col].str.replace('%',''
                                 ,regex=True).str.replace('$','',regex=True).str.replace(',',''
                                 ,regex=True).astype('float64')
        except:
            continue

    
    # find columns which contrain lists and dates 
    columns_lists = ['amenities','host_verifications']
    columns_dates = ['host_since','first_review', 'last_review']
    
    # replace all date columns by year and convert to int 
    for col in columns_dates:
        year_col = get_year_from_date(listings_clean, col)
        new_col_name = col + '_year'
        
        listings_clean[new_col_name] = year_col.astype("Int64")
        listings_clean = listings_clean.drop(col, axis=1)
        
        column_names.append(new_col_name)
    
    #extract all numerical and categorical columns
    columns_num = listings_clean.select_dtypes(include=['int64','float64']).columns.values
    columns_cat = listings_clean.select_dtypes(include='object').columns.values
    
    #leave only relevant columns by using pre-defined column list
    columns_num_relevant = list(np.intersect1d(columns_num,column_names))
    columns_cat_relevant = list(np.intersect1d(columns_cat,column_names))
    
    columns_rel = columns_num_relevant + columns_cat_relevant
    listings_rel_cols = listings_clean[columns_rel] 
                
    
    # create dummy columns for categorical variables
    for col in columns_cat_relevant:
        if col not in columns_lists:
            listings_rel_cols = pd.concat([listings_rel_cols
                              , pd.get_dummies(listings_rel_cols[col], dummy_na = False
                              , prefix = col, prefix_sep = "_", drop_first=False)], axis=1)    
        else:
            col_df = clean_str_to_list(listings_clean, col)
            listings_rel_cols = pd.concat([listings_rel_cols, col_df], axis=1)
        
        listings_rel_cols = listings_rel_cols.drop(col, axis=1)
    
    #handle null values
    listings_rel_cols = handle_null_values(listings_rel_cols)
    
    
    return listings_rel_cols

Building the price prediction model¶

Using the information gained above we can create a function to that iterates through different numbers of features and plots it against the score of the model. Doing so we can determine how many features are useful to use in out modelling approach.

In [31]:
def build_model(df, num_features_value=0):
    '''
    INPUT:
    df - the DataFrame that the linear model is based on
    num_features_value - if 0 then we test different values for number of features; if > 0 we run model with
                        number of features equals num_features_value
    
    OUTPUT:
    pipe - the model pipeline including StandardScaler and RandomForestRegressor
    num_features - list of number of features
    scores - list of scores for the numbers of features
    
    Iterates over the num_features list and builds a model using the most correlated n features where
    n is the current number of features. We split the data into train and test sets with test size of 20%.
    Then we use a pipeline of StandardScaler and RandomForestRegressor to fit our model and calculate the 
    r2 score.
    '''
    
    if num_features_value == 0:
        
        num_features = []
        scores = []
        
        for i in range(2,df.shape[1]):
        
            high_corr_columns = list(np.abs(df.corr().loc['price']).sort_values(ascending=False)
                                     [1:i].index.values)
        
            X = df[high_corr_columns]
            y = df['price']
        
            X_train, X_test, y_train, y_test = train_test_split(X,y,test_size=.2, random_state=20)
            
            pipe = make_pipeline(StandardScaler(), RandomForestRegressor())#LogisticRegression())
            pipe.fit(X_train, y_train) 
        
            num_features.append(i)
            scores.append(pipe.score(X_test, y_test))
    
    
    else:
        high_corr_columns = list(np.abs(df.corr().loc['price']).sort_values(ascending=False)
                                 [1:num_features_value+1].index.values)
        
        X = df[high_corr_columns]
        y = df['price']
        
        X_train, X_test, y_train, y_test = train_test_split(X,y,test_size=.2, random_state=20)
            
        pipe = make_pipeline(StandardScaler(), RandomForestRegressor())#LogisticRegression())
        pipe.fit(X_train, y_train)  
        
        num_features = []
        scores = []
    
    return pipe, num_features, scores, X_test, y_test
